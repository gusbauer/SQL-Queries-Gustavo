/* 1. Seleccionar la localidad en la que se encuentra el departamento número 10.
      (Tabla LOCATIONS relacionada con DEPARTMENTS por LOCATION_ID) */

/* Paso 1: buscar el location_id del departamento 10 */
SELECT department_id,
       department_name,
       location_id
FROM departments
WHERE department_id = 10;

/* Paso 2: usar ese location_id para obtener la ciudad */
SELECT location_id,
       city AS localidad
FROM locations
WHERE location_id = 1700;   -- sustituir por el location_id encontrado


/* 2. ¿En qué país se encuentra localizado el departamento 30?
      (COUNTRIES se relaciona con LOCATIONS por COUNTRY_ID) */

/* Paso 1: location_id del departamento 30 */
SELECT department_id,
       department_name,
       location_id
FROM departments
WHERE department_id = 30;

/* Paso 2: country_id de ese location */
SELECT location_id,
       city,
       country_id
FROM locations
WHERE location_id = 1400;   -- sustituir por el location_id encontrado

/* Paso 3: nombre del país */
SELECT country_id,
       country_name
FROM countries
WHERE country_id = 'US';    -- sustituir por el country_id encontrado


/* 3. ¿En qué región se encuentra el departamento 30?
      (REGIONS se relaciona con COUNTRIES por REGION_ID) */

/* Paso 1: location_id del departamento 30 */
SELECT department_id,
       department_name,
       location_id
FROM departments
WHERE department_id = 30;

/* Paso 2: country_id de ese location */
SELECT location_id,
       city,
       country_id
FROM locations
WHERE location_id = 1400;   -- sustituir por el location_id encontrado

/* Paso 3: region_id del país */
SELECT country_id,
       country_name,
       region_id
FROM countries
WHERE country_id = 'US';    -- sustituir por el country_id encontrado

/* Paso 4: nombre de la región */
SELECT region_id,
       region_name
FROM regions
WHERE region_id = 2;        -- sustituir por el region_id encontrado


/* 4. ¿Qué departamentos están en la misma localidad que el departamento
      denominado ‘Benefits’? */

/* Paso 1: location_id del departamento 'Benefits' */
SELECT department_id,
       department_name,
       location_id
FROM departments
